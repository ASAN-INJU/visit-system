// 걸매리 주소 → 위도/경도 자동 입력용 Google Apps Script
// 사용법:
// 1) 구글시트에 주소 목록을 넣습니다.
// 2) 열 제목은 반드시: 이름, 주소, 구역, 위도, 경도
// 3) 확장 프로그램 → Apps Script → 이 코드 붙여넣기
// 4) save → runGeocode 실행

function runGeocode() {
  const sheet = SpreadsheetApp.getActiveSheet();
  const values = sheet.getDataRange().getValues();
  const header = values[0];

  const addressCol = header.indexOf("주소");
  const latCol = header.indexOf("위도");
  const lngCol = header.indexOf("경도");

  if (addressCol === -1 || latCol === -1 || lngCol === -1) {
    SpreadsheetApp.getUi().alert("주소, 위도, 경도 열 제목이 필요합니다.");
    return;
  }

  for (let r = 1; r < values.length; r++) {
    const address = values[r][addressCol];
    const oldLat = values[r][latCol];
    const oldLng = values[r][lngCol];

    if (!address || (oldLat && oldLng)) continue;

    const fullAddress = address.includes("충남") || address.includes("충청남도")
      ? address
      : "충청남도 아산시 인주면 " + address;

    try {
      const result = Maps.newGeocoder()
        .setRegion("kr")
        .setLanguage("ko")
        .geocode(fullAddress);

      if (result.status === "OK" && result.results.length > 0) {
        const loc = result.results[0].geometry.location;
        sheet.getRange(r + 1, latCol + 1).setValue(loc.lat);
        sheet.getRange(r + 1, lngCol + 1).setValue(loc.lng);
      } else {
        sheet.getRange(r + 1, latCol + 1).setValue("좌표확인필요");
        sheet.getRange(r + 1, lngCol + 1).setValue("");
      }

      Utilities.sleep(250); // 너무 빠른 요청 방지
    } catch (e) {
      sheet.getRange(r + 1, latCol + 1).setValue("오류");
      sheet.getRange(r + 1, lngCol + 1).setValue(e.message);
    }
  }

  SpreadsheetApp.getUi().alert("좌표 입력 작업이 완료되었습니다.");
}
